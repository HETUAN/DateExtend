## Javascript Date Extend. 

### Example
var date1 = new Date();
console.log(date1.Formate("yyyy-MM-dd hh:mm:ss.S"));

date1.AddYear(1);
console.log(date1.Formate("yyyy-MM-dd hh:mm:ss.S"));

date1.AddMinute(10);
console.log(date1.Formate("yyyy-MM-dd hh:mm:ss.S"));

var date2 = new Date();
console.log(date2.Compare(date2));
